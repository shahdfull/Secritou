## 2. Interpolations cassées (syntaxe single-brace vs double-brace)

i18next exige `{{variable}}` (double accolades). Les clés suivantes utilisent `{variable}` (simple accolade) — la variable n'est **jamais substituée**.

| Sévérité | Fichier JSON : ligne | Clé | Valeur défectueuse | Valeur correcte |

## 6. Formats localisés

### 6a. `format.ts` — locale toujours `"fr-FR"`

**Fichier** : `client/src/utils/format.ts:20, 36, 65, 84, 100, 137`

Toutes les fonctions (`formatCurrency`, `formatDate`, `formatNumber`, `formatDateTime`, `formatRelativeDate`, `formatPercent`) ont `locale = "fr-FR"` comme valeur par défaut **codée en dur**. Aucune ne lit `i18n.language`. En mode EN, les utilisateurs voient :

| Ce qu'un EN voit | Ce qu'il devrait voir |
|------------------|-----------------------|
| `04/07/2026` | `07/04/2026` |
| `1 234,50 TND` | `1,234.50 TND` |
| `il y a 2 jours` | `2 days ago` |
| `15,0 %` | `15.0%` |

**Correctif** :

```ts
// client/src/utils/format.ts — ajouter en haut du fichier
import i18n from "@/i18n";

function getLocale(): string {
  // i18next stocke "fr" ou "en" ; Intl.* attend "fr-FR" / "en-US"
  const lng = i18n.language ?? "fr";
  return lng === "en" ? "en-US" : "fr-FR";
}

// Remplacer `locale = "fr-FR"` par `locale = getLocale()` dans chaque signature :
export function formatCurrency(amount: ..., currency = "TND", locale = getLocale()): string
export function formatDate(date: ..., options?: ..., locale = getLocale()): string
// etc.
```

## 8. Script de détection CI

Créer le fichier suivant à la racine du projet :

```js
// scripts/check-i18n.mjs
/**
 * Vérifie la cohérence i18n :
 * 1. Clés utilisées dans le code mais absentes d'un fichier de traduction
 * 2. Interpolations single-brace {var} dans les JSON (doivent être {{var}})
 * 3. Clés définies dans un fichier mais absentes de l'autre (désynchronisation FR/EN)
 *
 * Usage : node scripts/check-i18n.mjs
 * Exit code 1 si des erreurs sont trouvées (compatible CI).
 */

import fs from "fs";
import path from "path";
import { globSync } from "glob";

// ── Config ──────────────────────────────────────────────────────────────────
const SRC_DIR   = "client/src";
const FR_JSON   = "client/src/i18n/locales/fr/translation.json";
const EN_JSON   = "client/src/i18n/locales/en/translation.json";
const GLOBS     = ["**/*.tsx", "**/*.ts"];
const IGNORE    = ["**/*.test.*", "**/node_modules/**", "**/*.json"];
// Keys that are intentionally dynamic (not extractable statically)
const DYNAMIC_PREFIXES = ["toasts.", "seo."];

// ── Helpers ──────────────────────────────────────────────────────────────────
function flattenKeys(obj, prefix = "") {
  return Object.entries(obj).flatMap(([k, v]) => {
    const full = prefix ? `${prefix}.${k}` : k;
    if (v !== null && typeof v === "object" && !Array.isArray(v)) {
      return flattenKeys(v, full);
    }
    return [{ key: full, value: typeof v === "string" ? v : JSON.stringify(v) }];
  });
}

function extractKeysFromSource(srcDir, globs, ignore) {
  const files = globs.flatMap(g =>
    globSync(g, { cwd: srcDir, ignore, absolute: true })
  );
  const keys = new Set();
  const KEY_RE = /\bt\(\s*["'`]([\w.]+)["'`]/g;

  for (const file of files) {
    const src = fs.readFileSync(file, "utf-8");
    for (const match of src.matchAll(KEY_RE)) {
      keys.add(match[1]);
    }
  }
  return keys;
}

function findSingleBraceInterpolations(entries) {
  return entries
    .filter(({ value }) => typeof value === "string" && /\{[a-zA-Z_]+\}/.test(value))
    .map(({ key, value }) => ({ key, value }));
}

// ── Main ─────────────────────────────────────────────────────────────────────
const frEntries = flattenKeys(JSON.parse(fs.readFileSync(FR_JSON, "utf-8")));
const enEntries = flattenKeys(JSON.parse(fs.readFileSync(EN_JSON, "utf-8")));
const frKeys    = new Set(frEntries.map(e => e.key));
const enKeys    = new Set(enEntries.map(e => e.key));
const usedKeys  = extractKeysFromSource(SRC_DIR, GLOBS, IGNORE);

let errors = 0;

// 1. Clés utilisées mais manquantes
console.log("\n── 1. CLÉS MANQUANTES ─────────────────────────────────────────");
for (const key of usedKeys) {
  if (DYNAMIC_PREFIXES.some(p => key.startsWith(p))) continue;
  const missingFr = !frKeys.has(key);
  const missingEn = !enKeys.has(key);
  if (missingFr || missingEn) {
    const where = [missingFr && "FR", missingEn && "EN"].filter(Boolean).join(", ");
    console.error(`  MISSING [${where}]  ${key}`);
    errors++;
  }
}

// 2. Interpolations single-brace (bug i18next)
console.log("\n── 2. INTERPOLATIONS SINGLE-BRACE (bug) ───────────────────────");
for (const bad of findSingleBraceInterpolations(frEntries)) {
  console.error(`  FR single-brace: "${bad.key}" → ${bad.value}`);
  errors++;
}
for (const bad of findSingleBraceInterpolations(enEntries)) {
  console.error(`  EN single-brace: "${bad.key}" → ${bad.value}`);
  errors++;
}

// 3. Désynchronisation FR ↔ EN
console.log("\n── 3. CLÉS DÉSYNCHRONISÉES FR ↔ EN ────────────────────────────");
for (const key of frKeys) {
  if (!enKeys.has(key)) {
    console.warn(`  IN FR, NOT EN:  ${key}`);
    errors++;
  }
}
for (const key of enKeys) {
  if (!frKeys.has(key)) {
    console.warn(`  IN EN, NOT FR:  ${key}`);
    errors++;
  }
}

// 4. Clés définies mais jamais utilisées (orphelines)
console.log("\n── 4. CLÉS ORPHELINES (définies, jamais appelées) ─────────────");
for (const key of frKeys) {
  if (!usedKeys.has(key) && !DYNAMIC_PREFIXES.some(p => key.startsWith(p))) {
    console.warn(`  ORPHAN FR: ${key}`);
  }
}

// ── Résultat ─────────────────────────────────────────────────────────────────
console.log(`\n── RÉSULTAT : ${errors} erreur(s) bloquante(s) ────────────────────`);
if (errors > 0) {
  console.error("❌ i18n check FAILED");
  process.exit(1);
} else {
  console.log("✅ i18n check passed");
}
```
